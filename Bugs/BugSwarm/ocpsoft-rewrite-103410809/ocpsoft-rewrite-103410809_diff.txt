+
+         if (address.isPathSet())
+            result.append(address.getPath());
+
+         if (address.isQuerySet())
+         {
+            if (address.isDomainSet() && !address.isPathSet())
+               result.append("/");
+            result.append('?').append(address.getQuery());

+            result.append('#').append(address.getAnchor());
+      }
