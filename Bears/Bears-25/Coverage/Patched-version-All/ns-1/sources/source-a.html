     public JavaType withHandlersFrom(JavaType src) {

         if (h != _typeHandler) {
             type = type.withTypeHandler(h);
         }
         h = src.getValueHandler();
         if (h != _valueHandler) {
             type = type.withValueHandler(h);
         }
         return type;
     }
